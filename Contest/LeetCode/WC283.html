class SolutionB {
    fun minimalKSum(nums: IntArray, k: Int): Long {
        var ans = 1L * k * (k + 1) / 2
        val n = nums.distinct().sorted()
        var cur = k
        for (i in n.indices) {
            if (n[i] <= cur) {
                cur++
                ans -= n[i]
                ans += cur
            }
        }
        return ans
    }
}

class SolutionC {
    fun createBinaryTree(descriptions: Array<IntArray>): TreeNode? {
        val seen = HashMap<Int, TreeNode>()
        val p = IntArray(100005)
        descriptions.forEach {
            val parent = seen.getOrDefault(it[0], TreeNode(it[0]))
            val child = seen.getOrDefault(it[1], TreeNode(it[1]))
            if (it[2] == 1) {
                parent.left = child
            } else {
                parent.right = child
            }
            p[it[0]]++
            p[it[1]] = -10000
            seen[it[0]] = parent
            seen[it[1]] = child
        }
        for (i in p.indices) {
            if (p[i] > 0) {
                return seen[i]
            }
        }
        return null
    }
}

class SolutionD {
    fun replaceNonCoprimes(nums: IntArray): List<Int> {
        val ans = Stack<Int>()
        for (i in nums.indices) {
            ans.push(nums[i])
            while (ans.size >= 2) {
                val a = ans.pop()
                val b = ans.pop()
                if (gcd(a, b) > 1) {
                    ans.push(lcm(a, b))
                } else {
                    ans.push(b)
                    ans.push(a)
                    break
                }
            }
        }
        return ans
    }
}
